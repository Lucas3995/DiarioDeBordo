## Task 5.1: Implement DiarioDeBordoDbContext and EF Core entity configuration

**Description:**  
Create `DiarioDeBordoDbContext` with the `Conteudo` DbSet (the only entity with a DbSet in Phase 2 — Dashboard and Feed are views, never persisted). Configure all entities using Fluent API in separate configuration classes. Use snake_case table and column names (PostgreSQL convention) via `UseSnakeCaseNamingConvention()`.

### `src/DiarioDeBordo.Infrastructure/Persistencia/DiarioDeBordoDbContext.cs`
```csharp
using Microsoft.EntityFrameworkCore;
using DiarioDeBordo.Infrastructure.Persistencia.Configuracoes;

namespace DiarioDeBordo.Infrastructure.Persistencia;

/// <summary>
/// DbContext do DiarioDeBordo.
/// REGRA: apenas Conteudo tem DbSet na Fase 2. Dashboard e Feed são views — nunca persistidas.
/// Outros BCs terão seus DbSets adicionados em fases futuras.
/// </summary>
public sealed class DiarioDeBordoDbContext : DbContext
{
    public DiarioDeBordoDbContext(DbContextOptions<DiarioDeBordoDbContext> options)
        : base(options) { }

    public DbSet<Core.Entidades.Conteudo> Conteudos => Set<Core.Entidades.Conteudo>();

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);

        // Apply all configurations from this assembly
        modelBuilder.ApplyConfigurationsFromAssembly(typeof(DiarioDeBordoDbContext).Assembly);
    }
}
```

using Microsoft.EntityFrameworkCore;
using Microsoft.EntityFrameworkCore.Metadata.Builders;
using DiarioDeBordo.Core.Entidades;
using DiarioDeBordo.Core.Enums;

namespace DiarioDeBordo.Infrastructure.Persistencia.Configuracoes;

internal sealed class ConteudoConfiguration : IEntityTypeConfiguration<Conteudo>
{
    public void Configure(EntityTypeBuilder<Conteudo> builder)
    {
        builder.ToTable("conteudos");

        builder.HasKey(c => c.Id);
        builder.Property(c => c.Id).HasColumnName("id").ValueGeneratedNever();

        builder.Property(c => c.Titulo)
            .HasColumnName("titulo")
            .HasMaxLength(500)
            .IsRequired();

        builder.Property(c => c.Descricao)
            .HasColumnName("descricao")
            .HasMaxLength(5000);

        builder.Property(c => c.Anotacoes)
            .HasColumnName("anotacoes")
            .HasMaxLength(10000);

        builder.Property(c => c.Nota)
            .HasColumnName("nota")
            .HasPrecision(4, 1); // 0.0 to 10.0

        builder.Property(c => c.Formato)
            .HasColumnName("formato")
            .HasConversion<string>() // Store enum as string for readability
            .HasMaxLength(50);

        builder.Property(c => c.Subtipo)
            .HasColumnName("subtipo")
            .HasMaxLength(100);

        builder.Property(c => c.Papel)
            .HasColumnName("papel")
            .HasConversion<string>()
            .HasMaxLength(50)
            .IsRequired();

        builder.Property(c => c.TipoColetaneaValor)
            .HasColumnName("tipo_coletanea")
            .HasConversion<string?>()
            .HasMaxLength(50);

        builder.Property(c => c.UsuarioId)
            .HasColumnName("usuario_id")
            .IsRequired();

        builder.Property(c => c.CriadoEm)
            .HasColumnName("criado_em")
            .IsRequired();

        builder.Property(c => c.AtualizadoEm)
            .HasColumnName("atualizado_em")
            .IsRequired();

        // Owned entity — Progresso (1:1, stored in same table)
        builder.OwnsOne(c => c.Progresso, progresso =>
        {
            progresso.Property(p => p.Estado)
                .HasColumnName("progresso_estado")
                .HasConversion<string>()
                .HasMaxLength(50)
                .IsRequired();

            progresso.Property(p => p.PosicaoAtual)
                .HasColumnName("progresso_posicao_atual")
                .HasMaxLength(500);

            progresso.Property(p => p.NotaManual)
                .HasColumnName("progresso_nota_manual")
                .HasMaxLength(5000);
        });

        // Indexes
        builder.HasIndex(c => c.UsuarioId)
            .HasDatabaseName("idx_conteudos_usuario_id");

        builder.HasIndex(c => new { c.UsuarioId, c.CriadoEm })
            .HasDatabaseName("idx_conteudos_usuario_criado_em");

        // Owned collections — Fontes and Imagens stored in separate tables
        builder.OwnsMany(c => c.Fontes, fonte =>
        {
            fonte.ToTable("fontes");
            fonte.HasKey(f => f.Id);
            fonte.Property(f => f.Id).HasColumnName("id").ValueGeneratedNever();
            fonte.Property(f => f.ConteudoId).HasColumnName("conteudo_id").IsRequired();
            fonte.Property(f => f.Tipo).HasColumnName("tipo").HasConversion<string>().HasMaxLength(50).IsRequired();
            fonte.Property(f => f.Valor).HasColumnName("valor").HasMaxLength(2000).IsRequired();
            fonte.Property(f => f.Plataforma).HasColumnName("plataforma").HasMaxLength(100);
            fonte.Property(f => f.Prioridade).HasColumnName("prioridade").IsRequired();

            fonte.HasIndex(f => new { f.ConteudoId, f.Prioridade })
                .IsUnique()
                .HasDatabaseName("idx_fontes_conteudo_prioridade_unique");
        });

        builder.OwnsMany(c => c.Imagens, imagem =>
        {
            imagem.ToTable("imagens_conteudo");
            imagem.HasKey(i => i.Id);
            imagem.Property(i => i.Id).HasColumnName("id").ValueGeneratedNever();
            imagem.Property(i => i.ConteudoId).HasColumnName("conteudo_id").IsRequired();
            imagem.Property(i => i.OrigemTipo).HasColumnName("origem_tipo").HasConversion<string>().HasMaxLength(50).IsRequired();
            imagem.Property(i => i.Caminho).HasColumnName("caminho").HasMaxLength(1000).IsRequired();
            imagem.Property(i => i.Principal).HasColumnName("principal").IsRequired();
        });
    }

## Task 5.2: Create ConteudoRepository and first EF Core migration

**Description:**  
Implement `ConteudoRepository` and `CategoriaRepository`. Then generate the first migration `InitialCreate`. Every query must filter by `usuarioId` — the security invariant from Padrões Técnicos v4 section 4.6.

**Files to create:**

### `src/DiarioDeBordo.Infrastructure/Repositorios/ConteudoRepository.cs`
```csharp
using DiarioDeBordo.Core.Entidades;
using DiarioDeBordo.Core.Repositorios;
using DiarioDeBordo.Infrastructure.Persistencia;
using Microsoft.EntityFrameworkCore;

namespace DiarioDeBordo.Infrastructure.Repositorios;

/// <summary>
/// Implementação do repositório de Conteudo via EF Core + PostgreSQL.
/// REGRA ABSOLUTA: toda query filtra por usuarioId — nunca acessa dados de outro usuário.
/// (Padrões Técnicos v4, seção 4.6)
/// </summary>
internal sealed class ConteudoRepository : IConteudoRepository
{
    private readonly DiarioDeBordoDbContext _context;

    public ConteudoRepository(DiarioDeBordoDbContext context)
    {
        _context = context;
    }

    public async Task AdicionarAsync(Conteudo conteudo, CancellationToken ct)
    {
        await _context.Conteudos.AddAsync(conteudo, ct);
        await _context.SaveChangesAsync(ct);
    }

    public async Task<Conteudo?> ObterPorIdAsync(Guid id, Guid usuarioId, CancellationToken ct)
    {
        return await _context.Conteudos
            .Where(c => c.Id == id && c.UsuarioId == usuarioId) // usuarioId MANDATORY
            .FirstOrDefaultAsync(ct);
    }

    public async Task AtualizarAsync(Conteudo conteudo, CancellationToken ct)
    {
        _context.Conteudos.Update(conteudo);
        await _context.SaveChangesAsync(ct);
    }

    public async Task RemoverAsync(Guid id, Guid usuarioId, CancellationToken ct)
    {
        var conteudo = await ObterPorIdAsync(id, usuarioId, ct);
        if (conteudo is not null)
        {
            _context.Conteudos.Remove(conteudo);
            await _context.SaveChangesAsync(ct);
        }
    }

    public async Task<Conteudo?> BuscarPorUrlFonteAsync(Guid usuarioId, string urlNormalizada, CancellationToken ct)
    {
        return await _context.Conteudos
            .Where(c => c.UsuarioId == usuarioId) // usuarioId MANDATORY
            .Where(c => c.Fontes.Any(f => f.Valor == urlNormalizada))
            .FirstOrDefaultAsync(ct);
    }

    public async Task<Conteudo?> BuscarPorIdentificadorFonteAsync(
        Guid usuarioId, string plataforma, string identificador, CancellationToken ct)
    {
        return await _context.Conteudos
            .Where(c => c.UsuarioId == usuarioId) // usuarioId MANDATORY
            .Where(c => c.Fontes.Any(f => f.Plataforma == plataforma && f.Valor == identificador))
            .FirstOrDefaultAsync(ct);
    }
}
```

### `src/DiarioDeBordo.Infrastructure/Repositorios/CategoriaRepository.cs`
```csharp
using DiarioDeBordo.Core.Entidades;
using DiarioDeBordo.Core.Repositorios;
using DiarioDeBordo.Infrastructure.Persistencia;
using Microsoft.EntityFrameworkCore;

namespace DiarioDeBordo.Infrastructure.Repositorios;

internal sealed class CategoriaRepository : ICategoriaRepository
{
    private readonly DiarioDeBordoDbContext _context;

    public CategoriaRepository(DiarioDeBordoDbContext context)
    {
        _context = context;
    }

    /// <summary>
    /// I-12: Operação atômica — retorna existente ou cria novo.
    /// NomeNormalizado é sempre lowercase para garantir unicidade case-insensitive.
    /// </summary>
    public async Task<Categoria> ObterOuCriarAsync(Guid usuarioId, string nome, CancellationToken ct)
    {
        var nomeNormalizado = nome.Trim().ToLowerInvariant();

        var existente = await _context.Categorias
            .Where(c => c.UsuarioId == usuarioId && c.NomeNormalizado == nomeNormalizado)
            .FirstOrDefaultAsync(ct);

        if (existente is not null)
            return existente;

        var nova = Categoria.Criar(usuarioId, nome);
        await _context.Categorias.AddAsync(nova, ct);
        await _context.SaveChangesAsync(ct);
        return nova;
    }

    public async Task<IReadOnlyList<Categoria>> ListarComAutocompletarAsync(
        Guid usuarioId, string prefixo, CancellationToken ct)
    {
        var prefixoNormalizado = prefixo.Trim().ToLowerInvariant();

        return await _context.Categorias
            .Where(c => c.UsuarioId == usuarioId
                        && c.NomeNormalizado.StartsWith(prefixoNormalizado))
            .OrderBy(c => c.Nome)
            .Take(20)
            .ToListAsync(ct);
    }
}
```

### `src/DiarioDeBordo.Infrastructure/Persistencia/DesignTimeDbContextFactory.cs`

Required for `dotnet ef migrations add` to work without running the application:
```csharp
using Microsoft.EntityFrameworkCore;
using Microsoft.EntityFrameworkCore.Design;

namespace DiarioDeBordo.Infrastructure.Persistencia;

/// <summary>
/// Usado pelo CLI do EF Core em design time (dotnet ef migrations add).
/// Não é usado em runtime — a conexão real vem do DI container.
/// </summary>
internal sealed class DesignTimeDbContextFactory : IDesignTimeDbContextFactory<DiarioDeBordoDbContext>
{
    public DiarioDeBordoDbContext CreateDbContext(string[] args)
    {
        var optionsBuilder = new DbContextOptionsBuilder<DiarioDeBordoDbContext>();
        optionsBuilder.UseNpgsql(
            $"Host=localhost;Port=15432;Database=diariodebordo_dev;Username=postgres;Password={Environment.GetEnvironmentVariable("DIARIODEBORDO_EF_DEV_PG_PASSWORD") ?? throw new InvalidOperationException("Set DIARIODEBORDO_EF_DEV_PG_PASSWORD env var for EF Core migrations. Run: export DIARIODEBORDO_EF_DEV_PG_PASSWORD=<dev_password>")}",
            npgsql => npgsql.UseAdminDatabase("postgres"));

        return new DiarioDeBordoDbContext(optionsBuilder.Build());
    }
}
```

### DI registration

`src/DiarioDeBordo.Infrastructure/DependencyInjection.cs`:
```csharp
using DiarioDeBordo.Core.Repositorios;
using DiarioDeBordo.Infrastructure.Persistencia;
using DiarioDeBordo.Infrastructure.Repositorios;
using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.DependencyInjection;

namespace DiarioDeBordo.Infrastructure;

public static class DependencyInjection
{
    public static IServiceCollection AddInfrastructure(
        this IServiceCollection services,
        string connectionString)
    {
        services.AddDbContext<DiarioDeBordoDbContext>(options =>
            options.UseNpgsql(connectionString));

        services.AddScoped<IConteudoRepository, ConteudoRepository>();
        services.AddScoped<ICategoriaRepository, CategoriaRepository>();

        return services;
    }
}
```
