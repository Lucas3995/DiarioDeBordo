## Task 7.1: Implement OS-native secure storage adapters

**Description:**  
Create `IArmazenamentoSeguro` implementations for Windows (DPAPI) and Linux (libsecret + fallback). Passwords are stored as bytes (never as strings). The fallback uses AES-256-GCM with a key derived from a machine-specific seed. Reference: Padrões Técnicos v4, seção 4.3.

### `src/DiarioDeBordo.Infrastructure/Seguranca/ArmazenamentoSeguroWindows.cs`
```csharp
#if WINDOWS || NET_WINDOWS
using System.Security.Cryptography;
using System.Text;
using DiarioDeBordo.Core.Infraestrutura;

namespace DiarioDeBordo.Infrastructure.Seguranca;

/// <summary>
/// Armazenamento seguro via DPAPI (Windows Data Protection API).
/// Escopo: CurrentUser — dados só acessíveis pelo mesmo usuário Windows que instalou.
/// (Padrões Técnicos v4, seção 4.3)
/// </summary>
internal sealed class ArmazenamentoSeguroWindows : IArmazenamentoSeguro
{
    private static readonly byte[] EntropiaApp =
        Encoding.UTF8.GetBytes("DiarioDeBordo.SecureStorage.v1");

    private static string ObterDiretorioBase()
    {
        var dir = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
            "DiarioDeBordo", "secrets");
        Directory.CreateDirectory(dir);
        return dir;
    }

    private static string ObterCaminho(string chave) =>
        Path.Combine(ObterDiretorioBase(), $"{Uri.EscapeDataString(chave)}.enc");

    public Task ArmazenarAsync(string chave, byte[] valor)
    {
        var criptografado = ProtectedData.Protect(valor, EntropiaApp, DataProtectionScope.CurrentUser);
        File.WriteAllBytes(ObterCaminho(chave), criptografado);
        return Task.CompletedTask;
    }

    public Task<byte[]?> RecuperarAsync(string chave)
    {
        var caminho = ObterCaminho(chave);
        if (!File.Exists(caminho)) return Task.FromResult<byte[]?>(null);

        var criptografado = File.ReadAllBytes(caminho);
        var decriptado = ProtectedData.Unprotect(criptografado, EntropiaApp, DataProtectionScope.CurrentUser);
        return Task.FromResult<byte[]?>(decriptado);
    }

    public Task RemoverAsync(string chave)
    {
        var caminho = ObterCaminho(chave);
        if (File.Exists(caminho)) File.Delete(caminho);
        return Task.CompletedTask;
    }
}
#endif
```

### `src/DiarioDeBordo.Infrastructure/Seguranca/ArmazenamentoSeguroLinux.cs`
```csharp
using System.Security.Cryptography;
using System.Text;
using DiarioDeBordo.Core.Infraestrutura;

namespace DiarioDeBordo.Infrastructure.Seguranca;

/// <summary>
/// Armazenamento seguro via libsecret/secret-tool (Linux — GNOME Keyring / KWallet).
/// Fallback para arquivo criptografado com AES-256-GCM se secret-tool não estiver disponível.
/// (Padrões Técnicos v4, seção 4.3)
/// </summary>
internal sealed class ArmazenamentoSeguroLinux : IArmazenamentoSeguro
{
    private const string AppAttribute = "diariodebordo";

    public async Task ArmazenarAsync(string chave, byte[] valor)
    {
        if (await SecretToolDisponivelAsync())
        {
            await ArmazenarViaSecretToolAsync(chave, valor);
        }
        else
        {
            await ArmazenarFallbackAsync(chave, valor);
        }
    }

    public async Task<byte[]?> RecuperarAsync(string chave)
    {
        if (await SecretToolDisponivelAsync())
            return await RecuperarViaSecretToolAsync(chave);

        return await RecuperarFallbackAsync(chave);
    }

    public async Task RemoverAsync(string chave)
    {
        if (await SecretToolDisponivelAsync())
        {
            using var process = new System.Diagnostics.Process();
            process.StartInfo = new System.Diagnostics.ProcessStartInfo
            {
                FileName = "secret-tool",
                ArgumentList = { "clear", "application", AppAttribute, "key", chave },
                UseShellExecute = false,
                CreateNoWindow = true,
            };
            process.Start();
            await process.WaitForExitAsync();
        }
        else
        {
            var caminho = ObterCaminhoFallback(chave);
            if (File.Exists(caminho)) File.Delete(caminho);
        }
    }

    private static async Task<bool> SecretToolDisponivelAsync()
    {
        try
        {
            using var process = new System.Diagnostics.Process();
            process.StartInfo = new System.Diagnostics.ProcessStartInfo
            {
                FileName = "which",
                ArgumentList = { "secret-tool" },
                UseShellExecute = false,
                CreateNoWindow = true,
                RedirectStandardOutput = true,
            };
            process.Start();
            await process.WaitForExitAsync();
            return process.ExitCode == 0;
        }
        catch
        {
            return false;
        }
    }

    private static async Task ArmazenarViaSecretToolAsync(string chave, byte[] valor)
    {
        var base64 = Convert.ToBase64String(valor);
        using var process = new System.Diagnostics.Process();
        process.StartInfo = new System.Diagnostics.ProcessStartInfo
        {
            FileName = "secret-tool",
            ArgumentList = { "store", "--label", $"DiarioDeBordo:{chave}",
                "application", AppAttribute, "key", chave },
            RedirectStandardInput = true,
            UseShellExecute = false,
            CreateNoWindow = true,
        };
        process.Start();
        await process.StandardInput.WriteAsync(base64);
        process.StandardInput.Close();
        await process.WaitForExitAsync();

        if (process.ExitCode != 0)
            throw new InvalidOperationException(
                $"secret-tool store falhou com exit code {process.ExitCode}");
    }

    private static async Task<byte[]?> RecuperarViaSecretToolAsync(string chave)
    {
        using var process = new System.Diagnostics.Process();
        process.StartInfo = new System.Diagnostics.ProcessStartInfo
        {
            FileName = "secret-tool",
            ArgumentList = { "lookup", "application", AppAttribute, "key", chave },
            RedirectStandardOutput = true,
            UseShellExecute = false,
            CreateNoWindow = true,
        };
        process.Start();
        var output = await process.StandardOutput.ReadToEndAsync();
        await process.WaitForExitAsync();

        if (process.ExitCode != 0 || string.IsNullOrEmpty(output))
            return null;

        return Convert.FromBase64String(output.Trim());
    }

    // ---- AES-256-GCM Fallback ----
    // Used when secret-tool is not available (headless environments, CI)

    private static string ObterCaminhoFallback(string chave)
    {
        var dir = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
            "DiarioDeBordo", "secrets");
        Directory.CreateDirectory(dir);
        // Linux: set directory permissions to 700
        System.IO.File.SetAttributes(dir, System.IO.FileAttributes.Normal);
        return Path.Combine(dir, $"{Uri.EscapeDataString(chave)}.aes");
    }

    private static async Task ArmazenarFallbackAsync(string chave, byte[] valor)
    {
        var chaveAes = ObterChaveFallback();
        var nonce = new byte[AesGcm.NonceByteSizes.MaxSize];
        RandomNumberGenerator.Fill(nonce);

        var tag = new byte[AesGcm.TagByteSizes.MaxSize];
        var ciphertext = new byte[valor.Length];

        using var aes = new AesGcm(chaveAes, tag.Length);
        aes.Encrypt(nonce, valor, ciphertext, tag);

        // Format: nonce (12 bytes) + tag (16 bytes) + ciphertext
        var armazenado = new byte[nonce.Length + tag.Length + ciphertext.Length];
        nonce.CopyTo(armazenado, 0);
        tag.CopyTo(armazenado, nonce.Length);
        ciphertext.CopyTo(armazenado, nonce.Length + tag.Length);

        await File.WriteAllBytesAsync(ObterCaminhoFallback(chave), armazenado);
    }

    private static async Task<byte[]?> RecuperarFallbackAsync(string chave)
    {
        var caminho = ObterCaminhoFallback(chave);
        if (!File.Exists(caminho)) return null;

        var armazenado = await File.ReadAllBytesAsync(caminho);
        var nonceSize = AesGcm.NonceByteSizes.MaxSize;
        var tagSize = AesGcm.TagByteSizes.MaxSize;

        var nonce = armazenado[..nonceSize];
        var tag = armazenado[nonceSize..(nonceSize + tagSize)];
        var ciphertext = armazenado[(nonceSize + tagSize)..];

        var plaintext = new byte[ciphertext.Length];
        var chaveAes = ObterChaveFallback();

        using var aes = new AesGcm(chaveAes, tag.Length);
        aes.Decrypt(nonce, ciphertext, tag, plaintext);

        return plaintext;
    }

    private static byte[] ObterChaveFallback()
    {
        // Machine-specific key derived from machine name + user name
        // Not ideal (derivable) but acceptable for fallback when no keyring is available
        var seed = Encoding.UTF8.GetBytes(
            $"DiarioDeBordo.FallbackKey.{Environment.MachineName}.{Environment.UserName}");
        return SHA256.HashData(seed);
    }
}
```

## Task 7.2: Implement PostgresBootstrap

**Description:**  
Implement `IPostgresBootstrap` using `pg_ctl` from bundled binaries. The bootstrap handles first-run `initdb`, subsequent starts, and migration execution. Port 15432 is hard-coded (ADR-002, D-01). Retry logic handles startup race condition. Reference: 02-RESEARCH.md section 2.2.

**Files to create:**

### `src/DiarioDeBordo.Infrastructure/Postgres/PostgresBootstrap.cs`
```csharp
using System.Diagnostics;
using System.Security.Cryptography;
using System.Text;
using DiarioDeBordo.Core.Infraestrutura;
using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Logging;
using DiarioDeBordo.Infrastructure.Persistencia;

namespace DiarioDeBordo.Infrastructure.Postgres;

/// <summary>
/// Gerencia o ciclo de vida do PostgreSQL bundled na porta 15432.
/// Sequência: IsInitialized? initdb → start → migrate.
/// (D-01; Padrões Técnicos v4, seção 3.3; 02-RESEARCH.md seção 2.2)
/// </summary>
public sealed class PostgresBootstrap : IPostgresBootstrap
{
    private const string Port = "15432";
    private const string DatabaseName = "diariodebordo";
    private const string UserName = "diariodebordo_app";

    private readonly string _pgBinDir;
    private readonly string _pgDataDir;
    private readonly IArmazenamentoSeguro _secureStorage;
    private readonly ILogger<PostgresBootstrap> _logger;
    private readonly Func<string, DiarioDeBordoDbContext> _contextFactory;

    public PostgresBootstrap(
        string pgBinDir,
        string pgDataDir,
        IArmazenamentoSeguro secureStorage,
        ILogger<PostgresBootstrap> logger,
        Func<string, DiarioDeBordoDbContext> contextFactory)
    {
        _pgBinDir = pgBinDir;
        _pgDataDir = pgDataDir;
        _secureStorage = secureStorage;
        _logger = logger;
        _contextFactory = contextFactory;
    }

    public async Task<bool> EnsureRunningAsync(CancellationToken ct)
    {
        _logger.LogInformation("PostgresBootstrap: verificando estado...");

        if (!await IsInitializedAsync())
        {
            _logger.LogInformation("PostgresBootstrap: primeiro uso — executando initdb...");
            await InitializeClusterAsync(ct);
        }

        if (!await IsRunningAsync())
        {
            _logger.LogInformation("PostgresBootstrap: iniciando processo PostgreSQL...");
            await StartAsync(ct);
        }

        _logger.LogInformation("PostgresBootstrap: aplicando migrations...");
        await RunMigrationsAsync(ct);

        _logger.LogInformation("PostgresBootstrap: pronto.");
        return true;
    }

    public Task<bool> IsInitializedAsync()
    {
        // PG_VERSION file indicates a valid initialized cluster
        var pgVersion = Path.Combine(_pgDataDir, "PG_VERSION");
        return Task.FromResult(File.Exists(pgVersion));
    }

    public async Task<bool> IsRunningAsync()
    {
        // pg_ctl status exits 0 if running, non-zero otherwise
        var result = await RunPgCommandAsync("pg_ctl",
            $"status -D \"{_pgDataDir}\"", CancellationToken.None, throwOnError: false);
        return result == 0;
    }

    private async Task InitializeClusterAsync(CancellationToken ct)
    {
        // 1. Generate strong password — use RandomNumberGenerator, NEVER System.Random
        var password = GenerateStrongPassword();
        await _secureStorage.ArmazenarAsync("pg_superuser_password",
            Encoding.UTF8.GetBytes(password));

        // 2. Write password file for initdb
        var pwFile = Path.GetTempFileName();
        try
        {
            await File.WriteAllTextAsync(pwFile, password, ct);

            // 3. initdb with data checksums (tamper detection)
            await RunPgCommandAsync("initdb",
                $"--pgdata=\"{_pgDataDir}\" " +
                $"--auth=scram-sha-256 " +
                $"--data-checksums " +
                $"--username=postgres " +
                $"--pwfile=\"{pwFile}\" " +
                $"--encoding=UTF8 " +
                $"--locale=pt_BR.UTF-8 " +
                "--no-instructions",
                ct);
        }
        finally
        {
            File.Delete(pwFile);
        }

        // 4. Configure postgresql.conf
        await ConfigurePostgresAsync();

        _logger.LogInformation("PostgresBootstrap: cluster inicializado com sucesso.");
    }

    private async Task ConfigurePostgresAsync()
    {
        var conf = Path.Combine(_pgDataDir, "postgresql.conf");
        var additionalConfig = new[]
        {
            $"port = {Port}",
            "listen_addresses = 'localhost'",       // localhost only — security (DA6)
            "log_destination = 'stderr'",
            "logging_collector = off",
            "log_statement = 'none'",               // NEVER log queries in production (DA6)
            "shared_buffers = 128MB",
            "work_mem = 16MB",
            "maintenance_work_mem = 64MB",
            "effective_cache_size = 512MB",
            "max_connections = 10",                 // Desktop app, low concurrency
            "password_encryption = 'scram-sha-256'",
        };
        await File.AppendAllLinesAsync(conf, additionalConfig);

        // pg_hba.conf: scram-sha-256 for local connections only
        var hba = Path.Combine(_pgDataDir, "pg_hba.conf");
        await File.WriteAllTextAsync(hba,
            "# DiarioDeBordo — only localhost connections allowed\n" +
            "local all all scram-sha-256\n" +
            "host  all all 127.0.0.1/32 scram-sha-256\n" +
            "host  all all ::1/128      scram-sha-256\n");
    }

    private async Task StartAsync(CancellationToken ct)
    {
        var logFile = Path.Combine(_pgDataDir, "postgres.log");
        await RunPgCommandAsync("pg_ctl",
            $"start -D \"{_pgDataDir}\" -l \"{logFile}\" -w -t 30",
            ct);

        // Wait for PostgreSQL to accept connections
        await WaitForConnectionAsync(ct);
    }

    private async Task WaitForConnectionAsync(CancellationToken ct)
    {
        for (var attempt = 1; attempt <= 10; attempt++)
        {
            var ready = await RunPgCommandAsync("pg_isready",
                $"-h localhost -p {Port}",
                ct, throwOnError: false);

            if (ready == 0) return;

            _logger.LogDebug("PostgresBootstrap: aguardando pg aceitar conexões (tentativa {N})...", attempt);
            await Task.Delay(TimeSpan.FromSeconds(1), ct);
        }

        throw new InvalidOperationException(
            $"PostgreSQL não aceitou conexões na porta {Port} após 10 tentativas.");
    }

    private async Task RunMigrationsAsync(CancellationToken ct)
    {
        var connectionString = await BuildConnectionStringAsync();
        var context = _contextFactory(connectionString);

        // Retry for startup race condition (Meurice et al. 2015, MSR)
        for (var attempt = 1; attempt <= 5; attempt++)
        {
            try
            {
                await context.Database.MigrateAsync(ct);
                return;
            }
            catch (Exception ex) when (attempt < 5)
            {
                _logger.LogWarning(ex,
                    "PostgresBootstrap: migration falhou (tentativa {N}), retentando...", attempt);
                await Task.Delay(TimeSpan.FromSeconds(Math.Pow(2, attempt)), ct);
            }
        }
    }

    private async Task<string> BuildConnectionStringAsync()
    {
        var passwordBytes = await _secureStorage.RecuperarAsync("pg_superuser_password")
            ?? throw new InvalidOperationException("Senha do PostgreSQL não encontrada no Secure Storage.");
        var password = Encoding.UTF8.GetString(passwordBytes);

        return $"Host=localhost;Port={Port};Database={DatabaseName};Username=postgres;Password={password}";
    }

    private async Task<int> RunPgCommandAsync(
        string executable, string arguments, CancellationToken ct, bool throwOnError = true)
    {
        var fullPath = Path.Combine(_pgBinDir, executable);

        // On Linux, no extension needed. On Windows, add .exe.
        if (OperatingSystem.IsWindows() && !fullPath.EndsWith(".exe", StringComparison.OrdinalIgnoreCase))
            fullPath += ".exe";

        using var process = new Process();
        process.StartInfo = new ProcessStartInfo
        {
            FileName = fullPath,
            Arguments = arguments,
            UseShellExecute = false,
            CreateNoWindow = true,
            RedirectStandardOutput = true,
            RedirectStandardError = true,
        };

        process.Start();
        var stdout = await process.StandardOutput.ReadToEndAsync(ct);
        var stderr = await process.StandardError.ReadToEndAsync(ct);
        await process.WaitForExitAsync(ct);

        if (throwOnError && process.ExitCode != 0)
        {
            _logger.LogError(
                "Comando PostgreSQL falhou: {Executable} {Args}\nStdout: {Stdout}\nStderr: {Stderr}",
                executable, arguments, stdout, stderr);
            throw new InvalidOperationException(
                $"'{executable}' falhou com exit code {process.ExitCode}. Stderr: {stderr}");
        }

        return process.ExitCode;
    }

    /// <summary>
    /// Gera senha forte usando RandomNumberGenerator (NUNCA System.Random).
    /// (Padrões Técnicos v4, seção 4.1 — DA4)
    /// </summary>
    private static string GenerateStrongPassword()
    {
        const string chars =
            "abcdefghijklmnopqrstuvwxyz" +
            "ABCDEFGHIJKLMNOPQRSTUVWXYZ" +
            "0123456789" +
            "!@#$%^&*";

        var bytes = new byte[32];
        RandomNumberGenerator.Fill(bytes);
        return new string(bytes.Select(b => chars[b % chars.Length]).ToArray());
    }
}
```

### `src/DiarioDeBordo.Infrastructure/Postgres/PostgresBootstrapFactory.cs`
```csharp
using DiarioDeBordo.Core.Infraestrutura;
using DiarioDeBordo.Infrastructure.Persistencia;
using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Logging;

namespace DiarioDeBordo.Infrastructure.Postgres;

/// <summary>Cria uma instância de PostgresBootstrap com os caminhos corretos para a plataforma.</summary>
public static class PostgresBootstrapFactory
{
    public static PostgresBootstrap Create(
        IArmazenamentoSeguro secureStorage,
        ILogger<PostgresBootstrap> logger)
    {
        var pgBinDir = ObterPgBinDir();
        var pgDataDir = ObterPgDataDir();

        return new PostgresBootstrap(
            pgBinDir, pgDataDir, secureStorage, logger,
            connectionString =>
            {
                var options = new DbContextOptionsBuilder<DiarioDeBordoDbContext>()
                    .UseNpgsql(connectionString)
                    .Options;
                return new DiarioDeBordoDbContext(options);
            });
    }

    private static string ObterPgBinDir()
    {
        // installer/postgres/ is relative to the application's base directory
        var appDir = AppContext.BaseDirectory;
        var pgDir = Path.Combine(appDir, "installer", "postgres",
            OperatingSystem.IsWindows() ? "bin" : "bin");

        return pgDir;
    }

    private static string ObterPgDataDir()
    {
        var dataHome = OperatingSystem.IsWindows()
            ? Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData)
            : Environment.GetEnvironmentVariable("XDG_DATA_HOME")
              ?? Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.UserProfile),
                  ".local", "share");

        var pgDataDir = Path.Combine(dataHome, "DiarioDeBordo", "pgdata");
        Directory.CreateDirectory(pgDataDir);
        return pgDataDir;
    }
}
```

```csharp
using DiarioDeBordo.Infrastructure.Postgres;
using DiarioDeBordo.Infrastructure.Seguranca;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Logging;

namespace DiarioDeBordo.Infrastructure;

public static class DependencyInjectionBootstrap
{
    /// <summary>
    /// Registers secure storage and PostgresBootstrap for use in the pre-DI bootstrap phase.
    /// Call this on the minimal pre-container before building the full DI container.
    /// Used by App.axaml.cs before calling AddInfrastructure (which requires the real connection string).

            services.AddSingleton<IArmazenamentoSeguro, ArmazenamentoSeguroLinux>();

            PostgresBootstrapFactory.Create(
                sp.GetRequiredService<IArmazenamentoSeguro>(),
                sp.GetRequiredService<ILogger<PostgresBootstrap>>()));

        return services;
    }
}
```
