```csharp
using Avalonia;
using Avalonia.Controls.ApplicationLifetimes;
using Avalonia.Markup.Xaml;
using DiarioDeBordo.Core.Infraestrutura;
using DiarioDeBordo.Infrastructure;
using DiarioDeBordo.Module.Acervo;
using DiarioDeBordo.UI.ViewModels;
using DiarioDeBordo.UI.Views;
using MediatR;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Logging;

namespace DiarioDeBordo.Desktop;

public partial class App : Application
{
    private IServiceProvider? _services;

    public static AppBuilder BuildAvaloniaApp()
        => AppBuilder.Configure<App>()
            .UsePlatformDetect()
            .WithInterFont()
            .LogToTrace();

    public override void Initialize()
    {
        AvaloniaXamlLoader.Load(this);
    }

    public override async void OnFrameworkInitializationCompleted()
    {
        // FIX (Blocker 1): Get real connection string from bootstrap BEFORE building full DI container.
        // Bootstrap requires only secure storage and logging — build minimal pre-DI container first.
        // Reference: Padrões Técnicos v4, seção 4.4 (bootstrap sequence).

        // Step 1: Build minimal pre-DI container for bootstrap (logging + secure storage only)
        var preServices = new ServiceCollection();
        preServices.AddLogging(logging => logging.AddConsole().SetMinimumLevel(LogLevel.Information));
        if (OperatingSystem.IsWindows())
            preServices.AddSingleton<IArmazenamentoSeguro, ArmazenamentoSeguroWindows>();
        else
            preServices.AddSingleton<IArmazenamentoSeguro, ArmazenamentoSeguroLinux>();
        preServices.AddSingleton<IPostgresBootstrap, PostgresBootstrap>();
        var preContainer = preServices.BuildServiceProvider();

        // Step 2: Run bootstrap — initdb (first run), pg_ctl start, migrations
        var bootstrap = preContainer.GetRequiredService<IPostgresBootstrap>();
        string realConnectionString;
        try
        {
            await bootstrap.EnsureRunningAsync(CancellationToken.None);
            realConnectionString = await bootstrap.BuildConnectionStringAsync(CancellationToken.None);
        }
        catch (Exception ex)
        {
            var logger = preContainer.GetRequiredService<ILogger<App>>();
            logger.LogCritical(ex, "Falha crítica ao inicializar PostgreSQL.");
            // TODO: Show error dialog via Avalonia MessageBox
            return;
        }

        // Step 3: Build full DI container with the REAL connection string from secure storage
        var services = new ServiceCollection();
        ConfigureServices(services, realConnectionString);
        _services = services.BuildServiceProvider();

        if (ApplicationLifetime is IClassicDesktopStyleApplicationLifetime desktop)
        {
            desktop.MainWindow = new MainWindow
            {
                DataContext = _services.GetRequiredService<MainViewModel>()
            };
        }

        base.OnFrameworkInitializationCompleted();
    }

    private static void ConfigureServices(IServiceCollection services, string connectionString)
    {
        // Logging
        services.AddLogging(logging => logging.AddConsole().SetMinimumLevel(LogLevel.Information));

        // Infrastructure (repositories, DbContext) — real connection string from secure storage
        services.AddInfrastructure(connectionString);

        // MediatR — scan Module.Acervo assembly for handlers
        services.AddMediatR(cfg =>
        {
            cfg.RegisterServicesFromAssembly(typeof(Module.Acervo.Commands.CriarConteudoCommand).Assembly);
        });

        // Feature flags placeholder (Phase 2 — all flags off)
        services.AddSingleton<Core.FeatureFlags.IFeatureFlags, Core.FeatureFlags.FeatureFlagsPlaceholder>();

        // ViewModels
        services.AddTransient<MainViewModel>();
        services.AddTransient<AcervoViewModel>();
        services.AddTransient<CriarConteudoViewModel>();
    }
}
```

```csharp
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using DiarioDeBordo.Core.Eventos;
using DiarioDeBordo.Module.Acervo.DTOs;
using DiarioDeBordo.Module.Acervo.Queries;
using DiarioDeBordo.Module.Shared.Paginacao;
using MediatR;

namespace DiarioDeBordo.UI.ViewModels;

public partial class AcervoViewModel : ObservableObject, INotificationHandler<ConteudoCriadoNotification>
{
    private readonly IMediator _mediator;
    private readonly Func<CriarConteudoViewModel> _criarVmFactory;

    // Walking skeleton: hard-coded usuarioId until authentication is implemented (Phase 8)
    private static readonly Guid _usuarioIdTemporario = Guid.Parse("00000000-0000-0000-0000-000000000001");

    [ObservableProperty]
    private PaginatedList<ConteudoResumoDto> _conteudos = PaginatedList<ConteudoResumoDto>.Vazia();

    [ObservableProperty]
    private bool _isLoading;

    [ObservableProperty]
    private bool _mostrarFormularioCriar;

    [ObservableProperty]
    private CriarConteudoViewModel? _criarConteudoViewModel;

    public bool IsEmpty => !IsLoading && Conteudos.TotalItems == 0;

    public AcervoViewModel(IMediator mediator, Func<CriarConteudoViewModel> criarVmFactory)
    {
        _mediator = mediator;
        _criarVmFactory = criarVmFactory;
    }

    // Called by Avalonia lifecycle — load data when view appears
    public async Task InicializarAsync()
    {
        await CarregarConteudosAsync();
    }

    [RelayCommand]
    private async Task CarregarConteudosAsync()
    {
        IsLoading = true;
        try
        {
            var resultado = await _mediator.Send(
                new ListarConteudosQuery(_usuarioIdTemporario, PaginacaoParams.Padrao));

            if (resultado.IsSuccess)
                Conteudos = resultado.Value!;
        }
        finally
        {
            IsLoading = false;
            OnPropertyChanged(nameof(IsEmpty));
        }
    }

    [RelayCommand]
    private void AbrirFormularioCriar()
    {
        CriarConteudoViewModel = _criarVmFactory();
        CriarConteudoViewModel.ConteudoCriado += OnConteudoCriado;
        MostrarFormularioCriar = true;
    }

    [RelayCommand]
    private void FecharFormularioCriar()
    {
        MostrarFormularioCriar = false;
        CriarConteudoViewModel = null;
    }

    private async void OnConteudoCriado(object? sender, EventArgs e)
    {
        FecharFormularioCriar();
        await CarregarConteudosAsync();
    }

    // MediatR notification handler — called when any component creates a Conteudo
    public async Task Handle(ConteudoCriadoNotification notification, CancellationToken ct)
    {
        await Avalonia.Threading.Dispatcher.UIThread.InvokeAsync(async () =>
        {
            await CarregarConteudosAsync();
        });
    }
}
```

### `src/DiarioDeBordo.UI/ViewModels/CriarConteudoViewModel.cs`
```csharp
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using DiarioDeBordo.Module.Acervo.Commands;
using MediatR;

namespace DiarioDeBordo.UI.ViewModels;

public partial class CriarConteudoViewModel : ObservableObject
{
    private readonly IMediator _mediator;
    private static readonly Guid _usuarioIdTemporario = Guid.Parse("00000000-0000-0000-0000-000000000001");

    public event EventHandler? ConteudoCriado;

    [ObservableProperty]
    [NotifyPropertyChangedFor(nameof(PodeSalvar))]
    private string _titulo = string.Empty;

    [ObservableProperty]
    private string? _descricao;

    [ObservableProperty]
    private string? _anotacoes;

    // Progressive disclosure (D-03) — additional fields hidden by default
    [ObservableProperty]
    private bool _mostrarDetalhes;

    [ObservableProperty]
    private string? _mensagemErro;

    [ObservableProperty]
    private bool _isSalvando;

    public bool PodeSalvar => !string.IsNullOrWhiteSpace(Titulo) && !IsSalvando;

    public CriarConteudoViewModel(IMediator mediator)
    {
        _mediator = mediator;
    }

    [RelayCommand]
    private void ToggleDetalhes()
    {
        MostrarDetalhes = !MostrarDetalhes;
    }

    [RelayCommand(CanExecute = nameof(PodeSalvar))]
    private async Task SalvarAsync()
    {
        MensagemErro = null;
        IsSalvando = true;

        try
        {
            var resultado = await _mediator.Send(
                new CriarConteudoCommand(_usuarioIdTemporario, Titulo));

            if (resultado.IsSuccess)
            {
                ConteudoCriado?.Invoke(this, EventArgs.Empty);
            }
            else
            {
                MensagemErro = resultado.Error!.Mensagem;
            }
        }
        finally
        {
            IsSalvando = false;
        }
    }
}
```
